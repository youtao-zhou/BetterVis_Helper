13.1 At the Beginning

# library(BetterVis)

13.2 BetterVis_LinePlot_Chart

This function addresses the fundamental need for creating both standard and faceted line plots.

13.2.1 Data Input

# Create sample data
# In your project: data("BetterVis_LinePlot_Chart_example", package="BetterVis")
set.seed(123)
df_line <- expand.grid(
  Day = 1:10,
  Treatment = paste("Treat", 1:4),
  Group = c("G1", "G2")
)
df_line$value <- ave(rnorm(nrow(df_line)), df_line$Treatment, FUN = function(x) cumsum(x) + rnorm(1, 5))
example_colors <- c( "#4489C8", "#ED7E7A","#008F91", "#FFCD44")

13.2.2 Simple Example

# Note: eval=FALSE as the function is not provided in user files
BetterVis_LinePlot_Chart (
  df = df_line,
  x_var = "Day",
  y_var = "value",
  fill_var = "Treatment",
  colors = example_colors,
  x_title = "Day", y_title = "Value",
  legend_title = ""
)

13.2.3 Advanced Example

# Note: eval=FALSE as the function is not provided in user files
BetterVis_LinePlot_Chart (
  df = df_line,
  x_var = "Day",
  y_var = "value",
  fill_var = "Treatment",
  colors = example_colors,
  confidence = TRUE,
  x_title = "Day", y_title = "Value",
  legend_title = "", legend_position = "top",
  size_global_adjust = 1,
  facet = TRUE, facet_var = "Group", facet_background_color = "grey", facet_background_border=FALSE
)
